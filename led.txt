int ldr=0;
void setup()
{
  Serial.begin(9600);
  pinMode(A0,INPUT);
  pinMode(10, OUTPUT);
  pinMode(9,OUTPUT);
}

void loop()
{
ldr=analogRead(A0);
  Serial.print("LDR:");
  Serial.println(ldr);

  if(ldr<=19){
    digitalWrite(9,0);
    digitalWrite(10, 1);
  }
  else{
    digitalWrite(10,0);
    digitalWrite(9,1);
  }
}
